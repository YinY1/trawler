_F = TypeVar("_F", bound=Callable[..., Any])


def _wrap_xhs_call(func: _F) -> _F:
    """Translate xhs library exceptions to trawler's exception hierarchy.

    Mapping (spec §3.1.2) — except 顺序不可调,具体在前,RequestException 兜底最后:
      NeedVerifyError → CaptchaError
      IPBlockError    → IpBlockError   (库大写 P,项目小写 p)
      SignError       → RetryableError
      DataFetchError  → DataError
      RequestException→ RetryableError  (catch-all, ordered LAST)
    """
    @functools.wraps(func)
    async def wrapper(*args: Any, **kwargs: Any) -> Any:
        try:
            return await func(*args, **kwargs)
        except NeedVerifyError as e:
            raise CaptchaError(f"XHS captcha challenge: {e}") from e
        except IPBlockError as e:
            raise IpBlockError(f"XHS IP blocked: {e}") from e
        except SignError as e:
            raise RetryableError(f"XHS sign error: {e}") from e
        except DataFetchError as e:
            raise DataError(f"XHS data fetch error: {e}") from e
        except requests.RequestException as e:
            raise RetryableError(f"XHS network error: {e}") from e

    return wrapper  # type: ignore[return-value]
```

注意:顶部需加 `from typing import TypeVar`(已有 `from typing import Any`,改为 `from typing import Any, TypeVar`)。

用法: TRAWLER_DUMP=1 uv run python scripts/verify_xhs_business.py
需要: config/cookies.toml 配好有效 xhs cookie

验证 3 项(spec §6.1):
1. 4 业务方法返回结构字段名(parser 假设是否对位)
2. 签名适配对登录态请求够用性(观察 461/471)
3. xsec_source=pc_feed vs pc_share(downloader 第一层)

输出: console + /tmp/xhs_*_dump.jsonl
"""

from __future__ import annotations

import asyncio
import sys

from platforms.xiaohongshu.async_xhs_wrapper import AsyncXhsClient
from shared.config import load_config

# 一个公开的用户 ID(测试用,可换成 config 配置的订阅用户)
TEST_USER_ID = "61ea4b8a0000000010003c15"


async def main() -> int:
    config = load_config()
    cookie = config.xiaohongshu.auth.cookie
    if not cookie:
        print("ERROR: 未配置 xhs cookie(config.xiaohongshu.auth.cookie)", file=sys.stderr)
        return 1

    client = AsyncXhsClient(cookie=cookie)

    try:
        # 1. get_user_notes
        print("\n=== [1] get_user_notes ===")
        try:
            notes_data = await client.get_user_notes(TEST_USER_ID)
            print(f"keys: {list(notes_data.keys())}")
            notes = notes_data.get("notes", [])
            print(f"notes count: {len(notes)}")
            if notes:
                first = notes[0]
                print(f"first note keys: {list(first.keys())}")
                print(f"  has display_title: {'display_title' in first}")
                print(f"  has cover: {'cover' in first}")
                print(f"  has interact_info: {'interact_info' in first}")
                print(f"  has xsec_token: {'xsec_token' in first}")
                print(f"  has note_id: {'note_id' in first}")

                first_note_id = first.get("note_id", "")
                first_xsec_token = first.get("xsec_token", "")
            else:
                print("  (no notes, skip downstream)")
                first_note_id, first_xsec_token = "", ""
        except Exception as e:
            print(f"FAIL: {type(e).__name__}: {e}")
            first_note_id, first_xsec_token = "", ""

        # 2. get_note_by_id (pc_feed)
        if first_note_id:
            print("\n=== [2] get_note_by_id (pc_feed) ===")
            try:
                detail = await client.get_note_by_id(
                    first_note_id, xsec_token=first_xsec_token, xsec_source="pc_feed"
                )
                print(f"keys: {list(detail.keys())}")
                print(f"  has desc: {'desc' in detail}")
                print(f"  has video: {'video' in detail}")
                print(f"  has image_list: {'image_list' in detail}")
            except Exception as e:
                print(f"FAIL: {type(e).__name__}: {e}")

        # 3. get_note_by_id (pc_share, 对比)
        if first_note_id:
            print("\n=== [2b] get_note_by_id (pc_share) ===")
            try:
                detail_share = await client.get_note_by_id(
                    first_note_id, xsec_token=first_xsec_token, xsec_source="pc_share"
                )
                print(f"keys: {list(detail_share.keys())}")
                print(f"  (对比 pc_feed 看字段差异)")
            except Exception as e:
                print(f"FAIL: {type(e).__name__}: {e}")

        # 4. get_note_comments
        if first_note_id:
            print("\n=== [3] get_note_comments ===")
            try:
                comments_data = await client.get_note_comments(
                    first_note_id, xsec_token=first_xsec_token
                )
                print(f"keys: {list(comments_data.keys())}")
                comments = comments_data.get("comments", [])
                print(f"comments count: {len(comments)}")
                if comments:
                    c = comments[0]
                    print(f"first comment keys: {list(c.keys())}")
                    print(f"  has content: {'content' in c}")
                    print(f"  has user_info: {'user_info' in c}")
                    print(f"  has like_count: {'like_count' in c}")
            except Exception as e:
                print(f"FAIL: {type(e).__name__}: {e}")

        # 5. get_user_by_keyword
        print("\n=== [4] get_user_by_keyword ===")
        try:
            users_data = await client.get_user_by_keyword("测试")
            print(f"keys: {list(users_data.keys())}")
            users = users_data.get("users", [])
            print(f"users count: {len(users)}")
            if users:
                u = users[0]
                print(f"first user keys: {list(u.keys())}")
        except Exception as e:
            print(f"FAIL: {type(e).__name__}: {e}")

    finally:
        await client.close()

    print("\n=== DONE ===")
    print("检查 /tmp/xhs_*_dump.jsonl 获取完整响应")
    return 0


if __name__ == "__main__":
    sys.exit(asyncio.run(main()))
```
