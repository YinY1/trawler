**`api/auth.py` 核心函数**：

```python
def _hash_token(plain: str) -> str:
    """SHA-256 hex hash。Token 是高熵随机串，无需 argon2。"""
    return hashlib.sha256(plain.encode("utf-8")).hexdigest()


def _verify_token(plain: str, expected_hash: str) -> bool:
    """常量时间比对（hmac.compare_digest），防 timing attack。"""
    actual = _hash_token(plain)
    return hmac.compare_digest(actual, expected_hash)


async def require_token(request: Request) -> str:
    """FastAPI 依赖：校验 Authorization: Bearer <token>。

    - 无 header / 格式错误 / token 不匹配 → 401
    - 通过 → 返回 token name（供日志/审计）
    """
    auth = request.headers.get("authorization", "")
    if not auth.startswith("Bearer "):
        raise HTTPException(status_code=401, detail="invalid or missing token")
    plain = auth[len("Bearer "):]
    cfg = load_auth_config()
    for entry in cfg.api_tokens:
        if _verify_token(plain, entry.token_hash):
            return entry.name
    raise HTTPException(status_code=401, detail="invalid or missing token")


def create_token(name: str, auth_path: Path = AUTH_TOML_PATH) -> str:
    """生成新 token，hash 后存 data/auth.toml，返回明文（仅此一次）。"""
    plain = secrets.token_urlsafe(32)
    cfg = load_auth_config()
    # 同名覆盖（先删）
    cfg.api_tokens = [t for t in cfg.api_tokens if t.name != name]
    cfg.api_tokens.append(
        ApiTokenEntry(name=name, token_hash=_hash_token(plain), created_at=datetime.now(timezone.utc).timestamp())
    )
    save_auth_config(cfg)
    return plain


def revoke_token(name: str, auth_path: Path = AUTH_TOML_PATH) -> bool:
    """按 name 删除 token，返回是否删除成功。"""
    cfg = load_auth_config()
    before = len(cfg.api_tokens)
    cfg.api_tokens = [t for t in cfg.api_tokens if t.name != name]
    if len(cfg.api_tokens) == before:
        return False
    save_auth_config(cfg)
    return True
```

## T2: check 路由（run / status / stream）

**目标**：`POST /check/run`、`GET /check/status`、`GET /check/stream`，全部复用 web 端业务函数。互斥锁与 Web UI 共享。

### 依赖
T1（require_token、中间件豁免）。

### 决策点（spec 已定，plan 记录）

- **run 返回 202 + task_id**：用 `uuid.uuid4().hex`。Web UI 走同一锁但不返回 task_id（它走 SSE），API 返回 task_id 供 bot 轮询 `/check/status`。
- **run 冲突返回 409 + task_id**：取 `app.state.check_task` 存的 task_id（如有），fallback 一个占位 uuid。
- **SSE 鉴权**：`/check/stream` **不**经过 `Depends(require_token)`（FastAPI 依赖在 SSE 流开始前注入，header 可读），但实际上 SSE 客户端可能不便带 header（部分浏览器 EventSource API 限制）。决策：**SSE 走 header 鉴权**，与其它端点一致（spec §鉴权：API Token「`Authorization: Bearer`」）。Bot 用 curl/requests 完全可带 header。若未来浏览器 EventSource 需求出现再加 `?token=` query param fallback。plan 里明确：**首版 SSE 只支持 header 鉴权**。
- **错误 → 422**：`parse_since` 抛 `ValueError` → 422；非法 phase → 422（手动校验抛 `HTTPException(422)`，不用 `Phase[...]` KeyError，给清晰 message）。
- **状态字段**：返回 `running` / `processed_count` / `started_at` / `task_id` / `log_history`（前 50 条）。

from __future__ import annotations

import asyncio
import logging
import time
import uuid
from typing import Any

from fastapi import APIRouter, Depends, HTTPException, Request
from fastapi.responses import StreamingResponse

from api.auth import require_token
from api.schemas import CheckRunRequest, CheckRunResponse, CheckStatusResponse
from core.engine import PipelineEngine
from core.pipeline import run_check_once
from run_check import parse_since
from shared.config import load_config
from shared.message_store import MessageStore
from shared.protocols import Phase
from web.routes.check import make_log_callback  # 直接 import 复用

logger = logging.getLogger(__name__)
router = APIRouter()


@router.post("/check/run", response_model=CheckRunResponse, status_code=202)
async def check_run(
    body: CheckRunRequest,
    request: Request,
    _token_name: str = Depends(require_token),
) -> CheckRunResponse:
    state = request.app.state
    if state.check_running:
        # 409 with current task_id
        existing_task_id = getattr(state, "api_task_id", None) or ""
        raise HTTPException(
            status_code=409,
            detail={"status": "already_running", "task_id": existing_task_id},
        )

    is_manual = body.mode == "manual"
    target_phase: Phase | None = None
    since_ts: int | None = None

    # 同步预校验（与 web/routes/check.py:108-119 对齐）
    if is_manual:
        reset_phase_str = body.reset_phase or "summarized"
        try:
            target_phase = Phase[reset_phase_str.upper()]
        except KeyError as exc:
            raise HTTPException(status_code=422, detail=f"未知的重跑阶段: {reset_phase_str}") from exc
        if body.since:
            try:
                since_ts = parse_since(body.since)
            except ValueError as exc:
                raise HTTPException(status_code=422, detail=str(exc)) from exc

    task_id = uuid.uuid4().hex
    state.check_running = True
    state.check_processed_count = 0
    state.check_started_at = time.time()
    state.log_history.clear()
    state.api_task_id = task_id  # 新增字段，供 409 引用
    cb = make_log_callback(state)

    async def _run() -> None:
        try:
            config = await load_config()
            if is_manual:
                assert target_phase is not None
                platform_filter = None if body.platform in (None, "all", "") else body.platform
                store = MessageStore(config.general.data_dir)
                matched = store.query_messages(
                    since=since_ts,
                    title=body.title,
                    author=body.author,
                    platform=platform_filter,
                )
                if not matched:
                    cb("log", "⚠️ 没有匹配的消息")
                    cb("done", "✅ 手动重跑完成（无匹配）")
                    return
                cb("log", f"📋 匹配 {len(matched)} 条消息，从 {target_phase.name} 重跑")
                msg_ids = [m.msg_id for m in matched]
                await PipelineEngine.run_specific_messages(
                    msg_ids=msg_ids,
                    from_phase=target_phase,
                    skip_push=body.skip_push,
                    config=config,
                    store=store,
                    log_callback=cb,
                )
            else:
                await run_check_once(
                    config, platform=body.platform or "all", log_callback=cb
                )
        except Exception as exc:
            err_item = {
                "type": "error",
                "message": f"检查失败: {exc}",
                "time": time.strftime("%H:%M:%S"),
                "_ts": time.time(),
            }
            state.log_history.append(err_item)
            for sub in list(state.subscribers):
                try:
                    sub.put_nowait(err_item)
                except asyncio.QueueFull:
                    pass
        finally:
            state.check_running = False
            state.check_started_at = None
            state.check_task = None
            state.api_task_id = None
            for sub in list(state.subscribers):
                try:
                    sub.put_nowait(None)
                except asyncio.QueueFull:
                    pass

    state.check_task = asyncio.create_task(_run())
    return CheckRunResponse(status="started", task_id=task_id, mode="manual" if is_manual else "full")

#### SSE 重构（避免代码复制）

`web/routes/check.py:202-270` 的 `check_stream` 内部 generator 较长。决策：**抽出共享 helper**，不复制：

改 `web/routes/check.py`，把 `event_generator` 抽成模块级函数 `build_sse_response(state) -> StreamingResponse`，原 `check_stream` 路由调用它：

```python
# 在 web/routes/check.py 末尾追加

def build_sse_response(state: Any) -> StreamingResponse:
    """构造 SSE StreamingResponse（web 端 check_stream 和 api 端 check_stream 共用）。

    抽出避免复制 web/routes/check.py 的 generator 逻辑。generator body 完整搬自
    原 check_stream 内部（history_snapshot 过滤 / live 队列循环 / heartbeat / finally deregister），
    以保证 tests/test_web_check.py 行为等价。
    """
    sub_queue: asyncio.Queue[dict[str, object] | None] = asyncio.Queue(maxsize=2000)
    state.subscribers.append(sub_queue)
    # Snapshot history at connect time, but ONLY items from the current run.
    current_run_started = state.check_started_at
    history_snapshot = [
        item for item in state.log_history
        if not current_run_started or item.get("_ts", 0) >= current_run_started
    ]

    async def event_generator() -> AsyncIterator[bytes]:
        try:
            # 1) Replay history captured at connect time (current run only)
            for item in history_snapshot:
                clean = {k: v for k, v in item.items() if k != "_ts"}
                data = json.dumps(clean, ensure_ascii=False)
                yield f"event: log\ndata: {data}\n\n".encode("utf-8")
            # 2) If a finished run's history was already present at connect
            #    time, emit done immediately so the client UI settles.
            if not state.check_running and history_snapshot:
                yield b"event: done\ndata: \n\n"
                return
            # 3) Stream live events from our private queue
            while True:
                try:
                    item = await asyncio.wait_for(sub_queue.get(), timeout=30)
                except asyncio.TimeoutError:
                    yield b": heartbeat\n\n"  # SSE keepalive comment
                    continue
                if item is None:
                    yield b"event: done\ndata: \n\n"
                    return
                # Strip internal _ts so the payload stays clean.
                clean = {k: v for k, v in item.items() if k != "_ts"}
                data = json.dumps(clean, ensure_ascii=False)
                yield f"event: log\ndata: {data}\n\n".encode("utf-8")
        finally:
            # Always deregister so producer never queues into a dead queue.
            try:
                state.subscribers.remove(sub_queue)
            except ValueError:
                pass

    return StreamingResponse(event_generator(), media_type="text/event-stream")


@router.get("/check/stream")
async def check_stream(request: Request) -> StreamingResponse:
    """SSE endpoint（Web UI 用）。"""
    return build_sse_response(request.app.state)
```

**实施动作**：把原 `check_stream` 函数体（含 `sub_queue` / `history_snapshot` / `event_generator` / `return StreamingResponse(...)` 整段，约 `web/routes/check.py:202-270`）替换成上面的 `build_sse_response` helper + 1 行路由。`api/routes/check.py` 的 `check_stream` 直接 `return build_sse_response(request.app.state)`。

## T3: messages 路由（list / get / rerun）

**目标**：`GET /messages`、`GET /messages/{msg_id}`、`POST /messages/rerun`。

### 依赖
T1。

### 决策点

- **rerun 后台 task**：与 `/check/run` 一致，后台 `asyncio.create_task` + 立即返回 task_id。
- **rerun 走 `state.check_running` 单锁**（与 `/check/run` 完全对称，**有意为之**）：现有 `web/routes/messages.py` 的 `POST /messages/batch-reprocess` 明确获取并持有 `state.check_running` 锁，与 `/check/run` 互斥，防止 detector 与 rerun 并发写同一 store。API 端 `POST /messages/rerun` 必须遵循同一不变量：
  - 占锁前检查：已运行 → 返回 409 + `{"status": "already_running", "task_id": "..."}`
  - 成功占锁 → 立即返回 202 + task_id（uuid4）
  - 后台 task `finally` 块清理：`state.check_running=False`、`state.check_started_at=None`、`state.check_task=None`、广播 EOF（None）到 `state.subscribers`
  - 这意味着 `/messages/rerun` 与 `/check/run` **互斥**，两边 API / Web UI 都互斥
- **`since` 参数支持两种**：`parse_since` 字符串格式（`24h`/`7d`/`2026-06-01`）或 unix 时间戳整数。query 参数统一以字符串接收，先尝试 int 解析，失败走 `parse_since`。

@router.post("/messages/rerun", response_model=RerunResponse, status_code=202)
async def rerun_messages(
    body: RerunRequest,
    request: Request,
    _token_name: str = Depends(require_token),
) -> RerunResponse:
    """批量重跑指定消息。

    走 ``state.check_running`` 单锁（与 ``/check/run`` 及 web 端
    ``/messages/batch-reprocess`` 完全对称、互斥），防止 detector 与 rerun
    并发写同一 store。详见 T3 决策点。
    """
    try:
        target_phase = Phase[body.from_phase.upper()]
    except KeyError as exc:
        raise HTTPException(status_code=422, detail=f"未知 phase: {body.from_phase}") from exc

    state = request.app.state
    # 占锁前检查（与 /check/run 对称）
    if state.check_running:
        existing_task_id = getattr(state, "api_task_id", None) or ""
        raise HTTPException(
            status_code=409,
            detail={"status": "already_running", "task_id": existing_task_id},
        )

    cfg = await load_config()
    store = MessageStore(cfg.general.data_dir)
    # 实际 reset 数量：先查存在的消息数（reset_specific 在后台 task 里调）
    existing = [m for m in (store.get_message(mid) for mid in body.msg_ids) if m is not None]
    count = len(existing)
    if count == 0:
        raise HTTPException(status_code=404, detail="未找到任何指定消息")

    task_id = uuid.uuid4().hex
    # 占锁 + reset 状态（与 /check/run 完全对称）
    state.check_running = True
    state.check_processed_count = 0
    state.check_started_at = time.time()
    state.log_history.clear()
    state.api_task_id = task_id  # 供 409 引用
    cb = make_log_callback(state)

    async def _rerun() -> None:
        try:
            await PipelineEngine.run_specific_messages(
                msg_ids=body.msg_ids,
                from_phase=target_phase,
                skip_push=body.skip_push,
                config=cfg,
                store=store,
                log_callback=cb,  # 接入 SSE，与 /check/run 行为一致
            )
        except Exception as exc:
            err_item = {
                "type": "error",
                "message": f"批量重跑失败: {exc}",
                "time": time.strftime("%H:%M:%S"),
                "_ts": time.time(),
            }
            state.log_history.append(err_item)
            for sub in list(state.subscribers):
                try:
                    sub.put_nowait(err_item)
                except asyncio.QueueFull:
                    pass
        finally:
            # 与 /check/run 对称的清理
            state.check_running = False
            state.check_started_at = None
            state.check_task = None
            state.api_task_id = None
            for sub in list(state.subscribers):
                try:
                    sub.put_nowait(None)
                except asyncio.QueueFull:
                    pass

    state.check_task = asyncio.create_task(_rerun())
    return RerunResponse(status="started", task_id=task_id, count=count)
```

    parser = argparse.ArgumentParser(prog="api.token_tool", description="API token 管理")
    sub = parser.add_subparsers(dest="cmd", required=True)

    p_create = sub.add_parser("create", help="生成新 token，明文只打印一次")
    p_create.add_argument("name", help="token 名（如 bot-name）")

    sub.add_parser("list", help="列出所有 token（只显示 hash 前 8 位）")

    p_revoke = sub.add_parser("revoke", help="按名删除 token")
    p_revoke.add_argument("name")

    args = parser.parse_args(argv)

    if args.cmd == "create":
        plain = create_token(args.name)
        console.print(f"[green]✓ Token 创建成功（名: {args.name}）[/]")
        console.print(f"[bold yellow]⚠ 明文（仅此一次）: {plain}[/]")
        console.print("[dim]存储为 SHA-256 hash，无法恢复。[/]")
        return 0
    elif args.cmd == "list":
        cfg = load_auth_config()
        if not cfg.api_tokens:
            console.print("[dim]无 API token。[/]")
            return 0
        table = Table(title="API Tokens")
        table.add_column("Name")
        table.add_column("Hash (前 8 位)")
        table.add_column("Created At")
        for t in cfg.api_tokens:
            from datetime import datetime
            created = datetime.fromtimestamp(t.created_at).strftime("%Y-%m-%d %H:%M")
            table.add_row(t.name, t.token_hash[:8], created)
        console.print(table)
        return 0
    elif args.cmd == "revoke":
        if revoke_token(args.name):
            console.print(f"[green]✓ 已删除 token: {args.name}[/]")
            return 0
        else:
            console.print(f"[red]✗ 未找到 token: {args.name}[/]")
            return 1
    return 1


if __name__ == "__main__":
    sys.exit(main())
```

# Trawler HTTP API

## 概览

所有端点挂在 `/api/v1` 前缀，JSON in / JSON out。鉴权用 Bearer token（与 Web UI session 完全隔离）。

## 鉴权

1. 生成 token：
   ```bash
   python -m api.token_tool create my-bot
   # 输出明文（仅此一次），存到安全位置
   ```
2. 每次请求带 header：
   ```
   Authorization: Bearer <plaintext-token>
   ```
3. 撤销：
   ```bash
   python -m api.token_tool revoke my-bot
   ```

未带 token / token 无效 → 401 `{"detail": "invalid or missing token"}`。

## 端点

### `GET /api/v1/health`（无鉴权）
健康检查。

### `POST /api/v1/check/run`
触发检查。

**请求体**：
```json
{"mode": "full", "platform": "all"}
```
或手动模式：
```json
{"mode": "manual", "since": "24h", "title": "关键词", "reset_phase": "summarized", "skip_push": true}
```

**响应**（202）：
```json
{"status": "started", "task_id": "<uuid>", "mode": "full"}
```

**冲突**（409）：已有 run 在跑。

**curl 示例**：
```bash
curl -X POST http://localhost:8000/api/v1/check/run \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"mode": "full"}'
```

### `GET /api/v1/check/status`
查当前 run 状态。

### `GET /api/v1/check/stream`
SSE 日志流（header 带 token）。

### `GET /api/v1/messages`
查消息。query 参数：`since`（`24h` / `7d` / `2026-06-01` / unix ts）、`title`、`author`、`platform`、`phase`。

### `GET /api/v1/messages/{msg_id}`
单条消息详情。

### `POST /api/v1/messages/rerun`
批量重跑。body：`{"msg_ids": [...], "from_phase": "summarized", "skip_push": true}`。

### `GET /api/v1/subscriptions`
### `POST /api/v1/subscriptions`
### `DELETE /api/v1/subscriptions/{platform}/{identifier}`

## 错误格式

| 状态码 | 含义 |
|---|---|
| 401 | token 缺失/无效 |
| 404 | 资源不存在 |
| 409 | 冲突（重复添加、已有 run） |
| 422 | 参数校验失败 |
| 500 | 内部错误 |

响应体：`{"detail": "..."}`。
```
